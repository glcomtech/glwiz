21/// Provides utilities for executing system commands, including `sudo` and user-level operations.
22///
23/// This submodule contains functions to run commands with elevated privileges or as the current user,
24/// handling tasks like copying files or executing shell scripts. It is used across other submodules
25/// for operations requiring system-level access.
26///
27/// # Example
28/// ```
29/// use glwiz::functionality::commands::run_sudo_command;
30/// let result = run_sudo_command("echo", &["Hello, world!"]);
31/// assert!(result.is_ok());
32/// ```
33pub mod commands;
34
35/// Manages user and root configuration file setups, such as `.zshrc` and `.vimrc`.
36///
37/// This submodule handles copying configuration files to user and root directories, ensuring
38/// consistent environments. It supports idempotent operations by checking for existing files
39/// and prompting for overwrites.
40///
41/// # See Also
42/// - `user_cfg`: For managing user-specific configuration data.
43pub mod configs;
44
45/// Retrieves environment variables for system configuration.
46///
47/// This submodule provides functions to access environment variables like `USER` and `HOME`,
48/// which are critical for user-specific setups. It returns errors for unset variables to ensure
49/// robust error handling.
50pub mod env;
51
52/// Configures iptables firewall rules for network security.
53///
54/// This submodule sets up and applies iptables rules by copying configuration files and
55/// executing restore commands. It ensures existing rules are preserved or overwritten with
56/// user confirmation.
57pub mod iptables;
58
59/// Contains utility functions for common setup tasks, such as user input handling and license display.
60///
61/// This submodule provides functions for tasks like prompting users for software lists, validating
62/// root privileges, and displaying license information. It is used to enhance user interaction
63/// and setup logic.
64///
65/// # Example
66/// ```
67/// use glwiz::functionality::prog_fun::print_license_info;
68/// print_license_info(); // Displays the GNU GPL v3 license information
69/// ```
70pub mod prog_fun;
71
72/// Manages shell configurations, including Zsh setup and plugin installations.
73///
74/// This submodule handles changing the default shell to Zsh, installing Oh My Zsh, and adding
75/// plugins like autosuggestions and syntax highlighting. It ensures idempotent operations to
76/// avoid redundant setups.
77///
78/// # See Also
79/// - `configs`: For related configuration file management.
80pub mod shell;
81
82/// Handles software package installation across supported Linux distributions.
83///
84/// This submodule detects the Linux distribution and uses the appropriate package manager
85/// (e.g., `pacman`, `apt`, `dnf`) to install software. It supports both default and custom
86/// package lists.
87pub mod software;
88
89/// Tracks and validates the results of configuration tasks.
90///
91/// This submodule defines the `TaskResult` struct and functions to validate task outcomes,
92/// ensuring comprehensive error reporting for the setup process. It is used to collect and
93/// summarize task statuses.
94pub mod task;
95
96/// Stores and manages user configuration data, such as username and home directory.
97///
98/// This submodule provides the `UserCfg` struct to hold user-specific data, with methods to
99/// set and retrieve usernames and home directories. It is used for user-specific configurations
100/// like shell and Vim setups.
101///
102/// # See Also
103/// - `configs`: For applying user configuration files.
104pub mod user_cfg;
